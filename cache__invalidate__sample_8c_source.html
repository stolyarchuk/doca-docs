357 static doca_error_t allocate_dma_task(struct cache_invalidate_sample_state *state)
358 {
359  union doca_data user_data = {0};
360  DOCA_LOG_INFO("Allocating DMA task");
361 
362  /* Use doca_buf_inventory_buf_get_by_data to initialize the source buffer */
363  EXIT_ON_FAILURE(doca_buf_inventory_buf_get_by_data(state->inventory,
364  state->mmap,
365  state->buffer,
366  DMA_BUFFER_SIZE,
367  &state->dma_source));
368 
373  EXIT_ON_FAILURE(doca_buf_inventory_buf_get_by_addr(state->inventory,
374  state->mmap,
375  state->buffer + DMA_BUFFER_SIZE,
376  DMA_BUFFER_SIZE,
377  &state->dma_destination));
378 
379  EXIT_ON_FAILURE(doca_dma_task_memcpy_alloc_init(state->dma,
380  state->dma_source,
381  state->dma_destination,
382  user_data,
383  &state->dma_task));
384 
385  return DOCA_SUCCESS;
386 }

458 static void cleanup(struct cache_invalidate_sample_state *state)
459 {
460  if (state->dma_task != NULL)
461  doca_task_free(doca_dma_task_memcpy_as_task(state->dma_task));
462 
463  if (state->cache_invalidate_task != NULL)
464  doca_task_free(doca_mmap_advise_task_invalidate_cache_as_doca_task(state->cache_invalidate_task));
465 
466  /* A context must be stopped before it is destroyed */
467  if (state->dma_ctx != NULL)
468  (void)doca_ctx_stop(state->dma_ctx);
469 
470  if (state->mmap_advise_ctx != NULL)
471  (void)doca_ctx_stop(state->mmap_advise_ctx);
472 
473  /* All contexts must be destroyed before PE is destroyed. Context destroy disconnects it from the PE */
474  if (state->dma != NULL)
475  (void)doca_dma_destroy(state->dma);
476 
477  if (state->mmap_advise != NULL)
478  (void)doca_mmap_advise_destroy(state->mmap_advise);
479 
480  if (state->pe != NULL)
481  (void)doca_pe_destroy(state->pe);
482 
483  if (state->dma_source != NULL)
484  (void)doca_buf_dec_refcount(state->dma_source, NULL);
485 
486  if (state->dma_destination != NULL)
487  (void)doca_buf_dec_refcount(state->dma_destination, NULL);
488 
489  if (state->inventory != NULL) {
490  (void)doca_buf_inventory_stop(state->inventory);
491  (void)doca_buf_inventory_destroy(state->inventory);
492  }
493 
494  if (state->mmap != NULL) {
495  (void)doca_mmap_stop(state->mmap);
496  (void)doca_mmap_destroy(state->mmap);
497  }
498 
499  if (state->device != NULL)
500  (void)doca_dev_close(state->device);
501 
502  if (state->buffer != NULL)
503  free(state->buffer);
504 }

514 static doca_error_t run(struct cache_invalidate_sample_state *state)
515 {
516  memset(state, 0, sizeof(*state));
517 
518  EXIT_ON_FAILURE(allocate_buffer(state));
519  EXIT_ON_FAILURE(open_device(state));
520  EXIT_ON_FAILURE(create_mmap(state));
521  EXIT_ON_FAILURE(create_buf_inventory(state));
522  EXIT_ON_FAILURE(create_pe(state));
523  EXIT_ON_FAILURE(create_dma(state));
524  EXIT_ON_FAILURE(create_mmap_advise(state));
525  EXIT_ON_FAILURE(doca_ctx_start(state->dma_ctx));
526  EXIT_ON_FAILURE(doca_ctx_start(state->mmap_advise_ctx));
527  EXIT_ON_FAILURE(allocate_dma_task(state));
528  EXIT_ON_FAILURE(allocate_cache_invalidate_task(state));
529  EXIT_ON_FAILURE(submit_dma_task(state));
530  EXIT_ON_FAILURE(poll_for_completion(state));
531 
532  return DOCA_SUCCESS;
533 }
