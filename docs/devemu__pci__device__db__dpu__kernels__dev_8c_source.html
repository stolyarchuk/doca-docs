40 __dpa_rpc__ uint64_t init_app_ctx_rpc(doca_dpa_dev_devemu_pci_db_completion_t db_comp, doca_dpa_dev_devemu_pci_db_t db)
41 {
42  app_ctx.db_comp = db_comp;
43 
44  if (doca_dpa_dev_devemu_pci_db_completion_bind_db(app_ctx.db_comp, db) < 0) {

72 /*
73  * Doorbell handler wakes up everytime a DB is rang from Host
74  *
75  * @thread_arg [in]: Value provided on setup of DPA thread
76  */
77 __dpa_global__ void db_handler(uint64_t __attribute__((__unused__)) thread_arg)
78 {
79  doca_dpa_dev_devemu_pci_db_completion_t db_comp = app_ctx.db_comp;
80 
81  doca_dpa_dev_devemu_pci_db_completion_element_t comp_element;
82  if (doca_dpa_dev_devemu_pci_get_db_completion(db_comp, &comp_element) != 1) {
83  DOCA_DPA_DEV_LOG_INFO("No completion found\n");
84  doca_dpa_dev_thread_reschedule();
85  }
86 
87  uint32_t dbr_value;
88  doca_dpa_dev_devemu_pci_db_t db;
89  doca_dpa_dev_uintptr_t user_data;
90  doca_dpa_dev_devemu_pci_db_completion_element_get_db_properties(db_comp, comp_element, &db, &user_data);
91 
92  doca_dpa_dev_devemu_pci_db_completion_ack(db_comp, /*num_comp=*/1);
93  doca_dpa_dev_devemu_pci_db_completion_request_notification(db_comp);
94 
95  doca_dpa_dev_devemu_pci_db_request_notification(db);
96 
97  dbr_value = doca_dpa_dev_devemu_pci_db_get_value(db);
98  DOCA_DPA_DEV_LOG_INFO("Received Doorbell value is %u\n", dbr_value);
99 
100  doca_dpa_dev_thread_reschedule();
101 }
