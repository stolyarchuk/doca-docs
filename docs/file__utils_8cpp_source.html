35 template <typename Container>
36 Container load_file_bytes(std::string const &file_name)
37 {
38  static_assert(sizeof(typename Container::value_type) == sizeof(char),
39  "load_file_bytes requires container to contain byte sized objects");
40  std::ifstream file{file_name, std::ios::binary | std::ios::ate};
41  if (!file) {
42  throw std::runtime_error{"Unable to open file: \"" + file_name + "\""};
43  }
44 
45  Container result;
46  result.resize(file.tellg());
47 
48  file.seekg(0);
49  if (!file.read(reinterpret_cast<char *>(result.data()), result.size())) {
50  throw std::runtime_error{"Failed to read content of file: \"" + file_name + "\""};
51  }
52 
53  return result;
54 }
