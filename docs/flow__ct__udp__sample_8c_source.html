41 /*
42  * Create RSS pipe
43  *
44  * @port [in]: Pipe port
45  * @status [in]: User context for adding entry
46  * @pipe [out]: Created pipe pointer
47  * @return: DOCA_SUCCESS on success and DOCA_ERROR otherwise.
48  */
49 static doca_error_t create_rss_pipe(struct doca_flow_port *port,
50  struct entries_status *status,
51  struct doca_flow_pipe **pipe)
52 {
53  struct doca_flow_match match;
54  struct doca_flow_pipe_cfg *cfg;
55  struct doca_flow_fwd fwd;
56  uint16_t rss_queues[1];
57  doca_error_t result;
58 
59  memset(&match, 0, sizeof(match));
60  memset(&fwd, 0, sizeof(fwd));
61 
62  result = doca_flow_pipe_cfg_create(&cfg, port);
63  if (result != DOCA_SUCCESS) {
64  DOCA_LOG_ERR("Failed to create doca_flow_pipe_cfg: %s", doca_error_get_descr(result));
65  return result;
66  }
67 
68  result = set_flow_pipe_cfg(cfg, "RSS_PIPE", DOCA_FLOW_PIPE_BASIC, false);
69  if (result != DOCA_SUCCESS) {
70  DOCA_LOG_ERR("Failed to set doca_flow_pipe_cfg: %s", doca_error_get_descr(result));
71  goto destroy_pipe_cfg;
72  }
73  result = doca_flow_pipe_cfg_set_match(cfg, &match, NULL);
74  if (result != DOCA_SUCCESS) {
75  DOCA_LOG_ERR("Failed to set doca_flow_pipe_cfg match: %s", doca_error_get_descr(result));
76  goto destroy_pipe_cfg;
77  }
78 
79  /* RSS queue - send matched traffic to queue 0 */
80  rss_queues[0] = 0;
81  fwd.type = DOCA_FLOW_FWD_RSS;
82  fwd.rss_type = DOCA_FLOW_RESOURCE_TYPE_NON_SHARED;
83  fwd.rss.queues_array = rss_queues;
84  fwd.rss.outer_flags = DOCA_FLOW_RSS_IPV4 | DOCA_FLOW_RSS_UDP;
85  fwd.rss.nr_queues = 1;
86 
87  result = doca_flow_pipe_create(cfg, &fwd, NULL, pipe);
88  if (result != DOCA_SUCCESS) {
89  DOCA_LOG_ERR("Failed to create RSS pipe: %s", doca_error_get_descr(result));
90  goto destroy_pipe_cfg;
91  }
92  doca_flow_pipe_cfg_destroy(cfg);
93 
94  /* Match on any packet */
95  result = doca_flow_pipe_add_entry(0, *pipe, &match, NULL, NULL, &fwd, 0, status, NULL);
96  if (result != DOCA_SUCCESS) {
97  DOCA_LOG_ERR("Failed to add RSS pipe entry: %s", doca_error_get_descr(result));
98  return result;
99  }
100 
101  result = doca_flow_entries_process(port, 0, DEFAULT_TIMEOUT_US, 0);
102  if (result != DOCA_SUCCESS)
103  DOCA_LOG_ERR("Failed to process RSS entry: %s", doca_error_get_descr(result));
104 
105  return result;
106 
107 destroy_pipe_cfg:
108  doca_flow_pipe_cfg_destroy(cfg);
109  return result;
110 }

380 /*
381  * Parse UDP packet to update CT tables
382  *
383  * @packet [in]: Packet to parse
384  * @match_o [out]: Origin match struct to fill
385  * @match_r [out]: Reply match struct to fill
386  */
387 static void parse_packet(struct rte_mbuf *packet,
388  struct doca_flow_ct_match *match_o,
389  struct doca_flow_ct_match *match_r)
390 {
391  uint8_t *l4_hdr;
392  struct rte_ipv4_hdr *ipv4_hdr;
393  const struct rte_udp_hdr *udp_hdr;
394 
395  ipv4_hdr = rte_pktmbuf_mtod_offset(packet, struct rte_ipv4_hdr *, sizeof(struct rte_ether_hdr));
396 
397  match_o->ipv4.src_ip = ipv4_hdr->src_addr;
398  match_o->ipv4.dst_ip = ipv4_hdr->dst_addr;
399  match_r->ipv4.src_ip = match_o->ipv4.dst_ip;
400  match_r->ipv4.dst_ip = match_o->ipv4.src_ip;
401 
402  l4_hdr = (typeof(l4_hdr))ipv4_hdr + rte_ipv4_hdr_len(ipv4_hdr);
403  udp_hdr = (typeof(udp_hdr))l4_hdr;
404 
405  match_o->ipv4.l4_port.src_port = udp_hdr->src_port;
406  match_o->ipv4.l4_port.dst_port = udp_hdr->dst_port;
407  match_r->ipv4.l4_port.src_port = match_o->ipv4.l4_port.dst_port;
408  match_r->ipv4.l4_port.dst_port = match_o->ipv4.l4_port.src_port;
409 
410  match_o->ipv4.next_proto = DOCA_FLOW_PROTO_UDP;
411  match_r->ipv4.next_proto = DOCA_FLOW_PROTO_UDP;
412 }

414 /*
415  * Dequeue packets from DPDK queues, parse and update CT tables with new connection 5 tuple
416  *
417  * @port [in]: Port id to which an entry should be inserted
418  * @ct_queue [in]: DOCA Flow CT queue number
419  * @ct_status [in]: User context for adding CT entry
420  * @return: DOCA_SUCCESS on success and DOCA_ERROR otherwise.
421  */
422 static doca_error_t process_packets(struct doca_flow_port *port, uint16_t ct_queue, struct entries_status *ct_status)
423 {
424  struct rte_mbuf *packets[PACKET_BURST];
425  struct doca_flow_ct_match match_o;
426  struct doca_flow_ct_match match_r;
427  struct doca_flow_pipe_entry *entry;
428  uint32_t flags;
429  doca_error_t result;
430  int i, entries, nb_packets = 0;
431  bool conn_found = false;
432 
433  memset(&match_o, 0, sizeof(match_o));
434  memset(&match_r, 0, sizeof(match_r));
435 
436  nb_packets = rte_eth_rx_burst(0, 0, packets, PACKET_BURST);
437  if (nb_packets == 0) {
438  DOCA_LOG_INFO("Sample didn't receive packets to process");
439  return DOCA_ERROR_BAD_STATE;
440  }
441 
442  entries = 0;
443  DOCA_LOG_INFO("Sample received %d packets", nb_packets);
444  for (i = 0; i < PACKET_BURST && i < nb_packets; i++) {
445  parse_packet(packets[i], &match_o, &match_r);
446  flags = DOCA_FLOW_CT_ENTRY_FLAGS_ALLOC_ON_MISS | DOCA_FLOW_CT_ENTRY_FLAGS_DUP_FILTER_ORIGIN |
447  DOCA_FLOW_CT_ENTRY_FLAGS_DUP_FILTER_REPLY;
448  /* Allocate CT entry */
449  result = doca_flow_ct_entry_prepare(ct_queue,
450  NULL,
451  flags,
452  &match_o,
453  packets[i]->hash.rss,
454  &match_r,
455  packets[i]->hash.rss,
456  &entry,
457  &conn_found);
458  if (result != DOCA_SUCCESS) {
459  DOCA_LOG_ERR("Failed to prepare CT entry\n");
460  return result;
461  }
462 
463  if (!conn_found) {
464  flags = DOCA_FLOW_CT_ENTRY_FLAGS_NO_WAIT | DOCA_FLOW_CT_ENTRY_FLAGS_DIR_ORIGIN |
465  DOCA_FLOW_CT_ENTRY_FLAGS_DIR_REPLY;
466  result = doca_flow_ct_add_entry(ct_queue,
467  NULL,
468  flags,
469  &match_o,
470  &match_r,
471  NULL,
472  NULL,
473  0,
474  0,
475  0,
476  ct_status,
477  entry);
478  if (result != DOCA_SUCCESS) {
479  DOCA_LOG_ERR("Failed to add CT pipe an entry: %s", doca_error_get_descr(result));
480  return result;
481  }
482  entries++;
483  }
484  }
485 
486  while (ct_status->nb_processed != entries) {
487  result = doca_flow_ct_entries_process(port, ct_queue, 0, 0, NULL);
488  if (result != DOCA_SUCCESS) {
489  DOCA_LOG_ERR("Failed to process Flow CT entries: %s", doca_error_get_descr(result));
490  return result;
491  }
492 
493  if (ct_status->failure) {
494  DOCA_LOG_ERR("Flow CT entries process returned with a failure");
495  return DOCA_ERROR_BAD_STATE;
496  }
497  }
498 
499  return DOCA_SUCCESS;
500 }

502 /*
503  * Run flow_ct_udp sample
504  *
505  * @nb_queues [in]: number of queues the sample will use
506  * @ct_dev [in]: Flow CT device
507  * @return: DOCA_SUCCESS on success and DOCA_ERROR otherwise.
508  */
509 doca_error_t flow_ct_udp(uint16_t nb_queues, struct doca_dev *ct_dev)
510 {
511  const int nb_ports = 2, nb_entries = 6;
512  struct flow_resources resource;
513  uint32_t nr_shared_resources[SHARED_RESOURCE_NUM_VALUES] = {0};
514  struct doca_flow_pipe *rss_pipe, *encap_pipe, *count_pipe, *ct_pipe = NULL, *udp_pipe;
515  struct doca_flow_port *ports[nb_ports];
516  struct doca_flow_meta o_zone_mask, o_modify_mask, r_zone_mask, r_modify_mask;
517  struct doca_dev *dev_arr[nb_ports];
