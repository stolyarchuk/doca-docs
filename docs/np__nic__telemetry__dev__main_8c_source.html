51 /*
52  * Called on link or port info state change.
53  * This callback is used to configure port counters to query RX bytes on
54  *
55  * @return - void
56  */
57 void doca_pcc_dev_user_port_info_changed(uint32_t portid)
58 {
59  counter_ids[portid] = DOCA_PCC_DEV_GET_PORT_COUNTER_ID(portid, DOCA_PCC_DEV_NIC_COUNTER_TYPE_RX_BYTES, 0);
60  /* number of ports to initiate counters for */
61  ports_num = count_ports(doca_pcc_dev_get_logical_ports());
62  /* Configure counters to read */
63  doca_pcc_dev_nic_counters_config(counter_ids, ports_num, current_sampled_rx_bytes);
64  counters_started = 1;
65  __dpa_thread_fence(__DPA_MEMORY, __DPA_W, __DPA_W);
66 }

75  if (counters_started) {

77  doca_pcc_dev_nic_counters_sample();
78  *rx_256_bytes_p = current_sampled_rx_bytes[doca_pcc_np_dev_get_port_num(in)];
79  *rx_ts_p = doca_pcc_dev_get_timer_lo();
80  __dpa_thread_fence(__DPA_MEMORY, __DPA_W, __DPA_W);
81  }
82
