85 /*
86  * DMA Memcpy task completed callback
87  *
88  * @dma_task [in]: Completed task
89  * @task_user_data [in]: doca_data from the task
90  * @ctx_user_data [in]: doca_data from the context
91  */
92 static void dma_memcpy_completed_callback(struct doca_dma_task_memcpy *dma_task,
93  union doca_data task_user_data,
94  union doca_data ctx_user_data)
95 {
96  uint8_t expected_value = (uint8_t)task_user_data.u64;
97  struct pe_task_try_submit_sample_state *state = (struct pe_task_try_submit_sample_state *)ctx_user_data.ptr;
98 
99  state->base.num_completed_tasks++;
100 
105  (void)process_completed_dma_memcpy_task(dma_task, expected_value);
106 
107  /* The task is no longer required, therefore it can be freed */
108  (void)dma_task_free(dma_task);
109 }

111 /*
112  * Memcpy task error callback
113  *
114  * @dma_task [in]: failed task
115  * @task_user_data [in]: doca_data from the task
116  * @ctx_user_data [in]: doca_data from the context
117  */
118 static void dma_memcpy_error_callback(struct doca_dma_task_memcpy *dma_task,
119  union doca_data task_user_data,
120  union doca_data ctx_user_data)
121 {
122  struct pe_task_try_submit_sample_state *state = (struct pe_task_try_submit_sample_state *)ctx_user_data.ptr;
123  struct doca_task *task = doca_dma_task_memcpy_as_task(dma_task);
124 
125  (void)task_user_data;
126 
127  /* This sample defines that a task is completed even if it is completed with error */
128  state->base.num_completed_tasks++;
129 
130  DOCA_LOG_ERR("Task failed with status %s", doca_error_get_descr(doca_task_get_status(task)));
131 
132  /* The task is no longer required, therefore it can be freed */
133  (void)dma_task_free(dma_task);
134 }

142 static doca_error_t create_dma(struct pe_task_try_submit_sample_state *state)
143 {
144  union doca_data ctx_user_data = {0};
145 
146  DOCA_LOG_INFO("Creating DMA");
147 
148  EXIT_ON_FAILURE(doca_dma_create(state->base.device, &state->dma));
149  state->dma_ctx = doca_dma_as_ctx(state->dma);
150 
151  /* A context can only be connected to one PE (PE can run multiple contexts) */
152  EXIT_ON_FAILURE(doca_pe_connect_ctx(state->base.pe, state->dma_ctx));
153 
159  ctx_user_data.ptr = state;
160  EXIT_ON_FAILURE(doca_ctx_set_user_data(state->dma_ctx, ctx_user_data));
161 
162  EXIT_ON_FAILURE(doca_dma_task_memcpy_set_conf(state->dma,
163  dma_memcpy_completed_callback,
164  dma_memcpy_error_callback,
165  NUM_TASKS));
166 
167  return DOCA_SUCCESS;
168 }

177 static doca_error_t try_submit_dma_tasks(uint32_t num_tasks, struct doca_dma_task_memcpy **tasks)
178 {
179  uint32_t task_id = 0;
180 
181  DOCA_LOG_INFO("Submitting tasks");
182 
183  for (task_id = 0; task_id < num_tasks; task_id++) {
184  doca_error_t status = doca_task_try_submit(doca_dma_task_memcpy_as_task(tasks[task_id]));
185 
186  if (status != DOCA_SUCCESS) {
187  DOCA_LOG_ERR("Failed to submit task with status %s", doca_error_get_descr(status));
188 
193  doca_task_free(doca_dma_task_memcpy_as_task(tasks[task_id]));
194  return status;
195  }
196  }
197 
198  return DOCA_SUCCESS;
199 }

209 static void cleanup(struct pe_task_try_submit_sample_state *state)
210 {
211  /* A context must be stopped before it is destroyed */
212  if (state->dma_ctx != NULL)
213  (void)doca_ctx_stop(state->dma_ctx);
214 
215  /* All contexts must be destroyed before PE is destroyed. Context destroy disconnects it from the PE */
216  if (state->dma != NULL)
217  (void)doca_dma_destroy(state->dma);
218 
219  pe_sample_base_cleanup(&state->base);
220 }

230 static doca_error_t run(struct pe_task_try_submit_sample_state *state)
231 {
232  memset(state, 0, sizeof(*state));
233 
234  state->base.buffer_size = BUFFER_SIZE;
235  state->base.buf_inventory_size = BUF_INVENTORY_SIZE;
236 
237  EXIT_ON_FAILURE(allocate_buffer(&state->base));
238  EXIT_ON_FAILURE(open_device(&state->base));
239  EXIT_ON_FAILURE(create_mmap(&state->base));
240  EXIT_ON_FAILURE(create_buf_inventory(&state->base));
241  EXIT_ON_FAILURE(create_pe(&state->base));
242  EXIT_ON_FAILURE(create_dma(state));
243  EXIT_ON_FAILURE(doca_ctx_start(state->dma_ctx));
244  EXIT_ON_FAILURE(allocate_dma_tasks(&state->base, state->dma, NUM_TASKS, DMA_BUFFER_SIZE, state->tasks));
245  EXIT_ON_FAILURE(try_submit_dma_tasks(NUM_TASKS, state->tasks));
246  EXIT_ON_FAILURE(poll_for_completion(&state->base, NUM_TASKS));
247 
248  return DOCA_SUCCESS;
249 }
