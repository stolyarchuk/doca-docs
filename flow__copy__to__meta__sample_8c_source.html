42 /*
43  * Create DOCA Flow pipe with changeable match on meta data
44  *
45  * @port [in]: port of the pipe
46  * @port_id [in]: port ID of the pipe
47  * @pipe [out]: created pipe pointer
48  * @return: DOCA_SUCCESS on success and DOCA_ERROR otherwise.
49  */
50 static doca_error_t create_match_meta_pipe(struct doca_flow_port *port, int port_id, struct doca_flow_pipe **pipe)
51 {
52  struct doca_flow_match match;
53  struct doca_flow_match match_mask;
54  struct doca_flow_actions actions, *actions_arr[NB_ACTIONS_ARR];
55  struct doca_flow_fwd fwd;
56  struct doca_flow_pipe_cfg *pipe_cfg;
57  doca_error_t result;
58 
59  memset(&match_mask, 0, sizeof(match_mask));
60  memset(&match, 0, sizeof(match));
61  memset(&actions, 0, sizeof(actions));
62  memset(&fwd, 0, sizeof(fwd));
63 
64  /* set match_mask on meta */
65  match_mask.meta.u32[0] = UINT32_MAX;
66  match_mask.meta.u32[1] = META_DMAC_47_32_FIELD;
67 
68  actions_arr[0] = &actions;
69 
70  result = doca_flow_pipe_cfg_create(&pipe_cfg, port);
71  if (result != DOCA_SUCCESS) {
72  DOCA_LOG_ERR("Failed to create doca_flow_pipe_cfg: %s", doca_error_get_descr(result));
73  return result;
74  }
75 
76  result = set_flow_pipe_cfg(pipe_cfg, "MATCH_PIPE", DOCA_FLOW_PIPE_BASIC, false);
77  if (result != DOCA_SUCCESS) {
78  DOCA_LOG_ERR("Failed to set doca_flow_pipe_cfg: %s", doca_error_get_descr(result));
79  goto destroy_pipe_cfg;
80  }
81  result = doca_flow_pipe_cfg_set_match(pipe_cfg, &match, &match_mask);
82  if (result != DOCA_SUCCESS) {
83  DOCA_LOG_ERR("Failed to set doca_flow_pipe_cfg match: %s", doca_error_get_descr(result));
84  goto destroy_pipe_cfg;
85  }
86  result = doca_flow_pipe_cfg_set_actions(pipe_cfg, actions_arr, NULL, NULL, NB_ACTIONS_ARR);
87  if (result != DOCA_SUCCESS) {
88  DOCA_LOG_ERR("Failed to set doca_flow_pipe_cfg actions: %s", doca_error_get_descr(result));
89  goto destroy_pipe_cfg;
90  }
91 
92  /* forwarding traffic to other port */
93  fwd.type = DOCA_FLOW_FWD_PORT;
94  fwd.port_id = port_id ^ 1;
95 
96  result = doca_flow_pipe_create(pipe_cfg, &fwd, NULL, pipe);
97 destroy_pipe_cfg:
98  doca_flow_pipe_cfg_destroy(pipe_cfg);
99  return result;
100 }

130 /*
131  * Create DOCA Flow pipe with 5 tuple and src mac address match, and copy src mac action
132  *
133  * @port [in]: port of the pipe
134  * @match_meta_pipe [in]: pipe to forward the matched packets
135  * @pipe [out]: created pipe pointer
136  * @return: DOCA_SUCCESS on success and DOCA_ERROR otherwise.
137  */
138 static doca_error_t create_copy_to_meta_pipe(struct doca_flow_port *port,
139  struct doca_flow_pipe *match_meta_pipe,
140  struct doca_flow_pipe **pipe)
141 {
142  struct doca_flow_match match;
143  struct doca_flow_actions actions;
144  struct doca_flow_actions *actions_arr[NB_ACTIONS_ARR];
145  struct doca_flow_fwd fwd;
146  struct doca_flow_fwd fwd_miss;
147  struct doca_flow_pipe_cfg *pipe_cfg;
148  struct doca_flow_action_descs descs;
149  struct doca_flow_action_descs *descs_arr[NB_ACTIONS_ARR];
150  struct doca_flow_action_desc desc_array[NB_ACTION_DESC] = {0};
151  doca_error_t result;
152 
153  memset(&match, 0, sizeof(match));
154  memset(&actions, 0, sizeof(actions));
155  memset(&fwd, 0, sizeof(fwd));
156  memset(&fwd_miss, 0, sizeof(fwd_miss));
157  memset(&descs, 0, sizeof(descs));
158 
159  /* copy src mac to u32[0]-u32[1] */
160  desc_array[0].type = DOCA_FLOW_ACTION_COPY;
161  desc_array[0].field_op.src.field_string = "outer.eth.src_mac";
162  desc_array[0].field_op.src.bit_offset = 0;
163  desc_array[0].field_op.dst.field_string = "meta.data";
164  desc_array[0].field_op.dst.bit_offset = META_U32_BIT_OFFSET(0);
165  desc_array[0].field_op.width = 32;
166 
167  desc_array[1].type = DOCA_FLOW_ACTION_COPY;
168  desc_array[1].field_op.src.field_string = "outer.eth.src_mac";
169  desc_array[1].field_op.src.bit_offset = 32;
170  desc_array[1].field_op.dst.field_string = "meta.data";
171  desc_array[1].field_op.dst.bit_offset = META_U32_BIT_OFFSET(1);
172  desc_array[1].field_op.width = 16;
173 
174  actions_arr[0] = &actions;
175  descs_arr[0] = &descs;
176  descs.nb_action_desc = NB_ACTION_DESC;
177  descs.desc_array = desc_array;
178 
179  /* 5 tuple match */
180  match.parser_meta.outer_l4_type = DOCA_FLOW_L4_META_TCP;
181  match.parser_meta.outer_l3_type = DOCA_FLOW_L3_META_IPV4;
182  match.outer.l4_type_ext = DOCA_FLOW_L4_TYPE_EXT_TCP;
183  match.outer.l3_type = DOCA_FLOW_L3_TYPE_IP4;
184  match.outer.ip4.src_ip = 0xffffffff;
185  match.outer.ip4.dst_ip = 0xffffffff;
186  match.outer.tcp.l4_port.src_port = 0xffff;
187  match.outer.tcp.l4_port.dst_port = 0xffff;
188 
189  SET_MAC_ADDR(match.outer.eth.src_mac, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff);
190 
191  result = doca_flow_pipe_cfg_create(&pipe_cfg, port);
192  if (result != DOCA_SUCCESS) {
193  DOCA_LOG_ERR("Failed to create doca_flow_pipe_cfg: %s", doca_error_get_descr(result));
194  return result;
195  }
196 
197  result = set_flow_pipe_cfg(pipe_cfg, "COPY_TO_META_PIPE", DOCA_FLOW_PIPE_BASIC, true);
198  if (result != DOCA_SUCCESS) {
199  DOCA_LOG_ERR("Failed to set doca_flow_pipe_cfg: %s", doca_error_get_descr(result));
200  goto destroy_pipe_cfg;
201  }
202  result = doca_flow_pipe_cfg_set_match(pipe_cfg, &match, NULL);
203  if (result != DOCA_SUCCESS) {
204  DOCA_LOG_ERR("Failed to set doca_flow_pipe_cfg match: %s", doca_error_get_descr(result));
205  goto destroy_pipe_cfg;
206  }
207  result = doca_flow_pipe_cfg_set_actions(pipe_cfg, actions_arr, NULL, descs_arr, NB_ACTIONS_ARR);
208  if (result != DOCA_SUCCESS) {
209  DOCA_LOG_ERR("Failed to set doca_flow_pipe_cfg actions: %s", doca_error_get_descr(result));
210  goto destroy_pipe_cfg;
211  }
212 
213  fwd.type = DOCA_FLOW_FWD_PIPE;
214  fwd.next_pipe = match_meta_pipe;
215 
216  result = doca_flow_pipe_create(pipe_cfg, &fwd, NULL, pipe);
217 destroy_pipe_cfg:
218  doca_flow_pipe_cfg_destroy(pipe_cfg);
219  return result;
220 }

222 /*
223  * Add DOCA Flow pipe entry with example values to match
224  *
225  * @pipe [in]: pipe of the entry
226  * @status [in]: user context for adding entry
227  * @return: DOCA_SUCCESS on success and DOCA_ERROR otherwise.
228  */
229 static doca_error_t add_copy_to_meta_pipe_entry(struct doca_flow_pipe *pipe, struct entries_status *status)
230 {
231  struct doca_flow_match match;
232  struct doca_flow_actions actions;
233  struct doca_flow_pipe_entry *entry;
234  doca_error_t result;
235 
236  doca_be32_t dst_ip_addr = BE_IPV4_ADDR(8, 8, 8, 8);
237  doca_be32_t src_ip_addr = BE_IPV4_ADDR(1, 2, 3, 4);
238  doca_be16_t dst_port = rte_cpu_to_be_16(80);
239  doca_be16_t src_port = rte_cpu_to_be_16(1234);
240  uint8_t src_mac[] = {0xaa, 0xbb, 0xcc, 0xdd, 0xee, 0xff};
241 
242  memset(&match, 0, sizeof(match));
243  memset(&actions, 0, sizeof(actions));
244 
245  match.outer.ip4.dst_ip = dst_ip_addr;
246  match.outer.ip4.src_ip = src_ip_addr;
247  match.outer.tcp.l4_port.dst_port = dst_port;
248  match.outer.tcp.l4_port.src_port = src_port;
249 
250  SET_MAC_ADDR(match.outer.eth.src_mac, src_mac[0], src_mac[1], src_mac[2], src_mac[3], src_mac[4], src_mac[5]);
251 
252  result = doca_flow_pipe_add_entry(0, pipe, &match, &actions, NULL, NULL, 0, status, &entry);
253  if (result != DOCA_SUCCESS)
254  return result;
255 
256  return DOCA_SUCCESS;
257 }

259 /*
260  * Run flow_copy_to_meta sample
261  *
262  * @nb_queues [in]: number of queues the sample will use
263  * @return: DOCA_SUCCESS on success and DOCA_ERROR otherwise.
264  */
265 doca_error_t flow_copy_to_meta(int nb_queues)
266 {
267  const int nb_ports = 2;
268  struct flow_resources resource = {0};
269  uint32_t nr_shared_resources[SHARED_RESOURCE_NUM_VALUES] = {0};
270  struct doca_flow_port *ports[nb_ports];
271  struct doca_dev *dev_arr[nb_ports];
272  uint32_t actions_mem_size[nb_ports];
273  struct doca_flow_pipe *pipe;
274  struct doca_flow_pipe *match_meta_pipe;
275  struct entries_status status;
276  int num_of_entries = 2;
277  doca_error_t result;
278  int port_id;
279 
280  result = init_doca_flow(nb_queues, "vnf,hws", &resource, nr_shared_resources);
281  if (result != DOCA_SUCCESS) {
282  DOCA_LOG_ERR("Failed to init DOCA Flow: %s", doca_error_get_descr(result));
283  return result;
284  }
285 
286  memset(dev_arr, 0, sizeof(struct doca_dev *) * nb_ports);
287  ARRAY_INIT(actions_mem_size, ACTIONS_MEM_SIZE(nb_queues, num_of_entries));
288  result = init_doca_flow_ports(nb_ports, ports, true, dev_arr, actions_mem_size);
289  if (result != DOCA_SUCCESS) {
290  DOCA_LOG_ERR("Failed to init DOCA ports: %s", doca_error_get_descr(result));
291  doca_flow_destroy();
292  return result;
293  }
294 
295  for (port_id = 0; port_id < nb_ports; port_id++) {
296  memset(&status, 0, sizeof(status));
297 
298  result = create_match_meta_pipe(ports[port_id], port_id, &match_meta_pipe);
299  if (result != DOCA_SUCCESS) {
300  DOCA_LOG_ERR("Failed to create pipe: %s", doca_error_get_descr(result));
301  stop_doca_flow_ports(nb_ports, ports);
302  doca_flow_destroy();
303  return result;
304  }
305 
306  result = add_match_meta_pipe_entry(match_meta_pipe, &status);
307  if (result != DOCA_SUCCESS) {
308  DOCA_LOG_ERR("Failed to add entry: %s", doca_error_get_descr(result));
309  stop_doca_flow_ports(nb_ports, ports);
310  doca_flow_destroy();
311  return result;
312  }
313 
314  result = create_copy_to_meta_pipe(ports[port_id], match_meta_pipe, &pipe);
315  if (result != DOCA_SUCCESS) {
316  DOCA_LOG_ERR("Failed to create pipe: %s", doca_error_get_descr(result));
317  stop_doca_flow_ports(nb_ports, ports);
318  doca_flow_destroy();
319  return result;
320  }
321 
322  result = add_copy_to_meta_pipe_entry(pipe, &status);
323  if (result != DOCA_SUCCESS) {
324  DOCA_LOG_ERR("Failed to add entry: %s", doca_error_get_descr(result));
325  stop_doca_flow_ports(nb_ports, ports);
326  doca_flow_destroy();
327  return result;
328  }
329 
330  result = doca_flow_entries_process(ports[port_id], 0, DEFAULT_TIMEOUT_US, num_of_entries);
331  if (result != DOCA_SUCCESS) {
332  DOCA_LOG_ERR("Failed to process entries: %s", doca_error_get_descr(result));
333  stop_doca_flow_ports(nb_ports, ports);
334  doca_flow_destroy();
335  return result;
336  }
337 
338  if (status.nb_processed != num_of_entries || status.failure) {
339  DOCA_LOG_ERR("Failed to process entries");
340  stop_doca_flow_ports(nb_ports, ports);
341  doca_flow_destroy();
342  return DOCA_ERROR_BAD_STATE;
343  }
344  }
345 
346  DOCA_LOG_INFO("Wait few seconds for packets to arrive");
347  sleep(5);
348 
349  result = stop_doca_flow_ports(nb_ports, ports);
350  doca_flow_destroy();
351  return result;
352 }
