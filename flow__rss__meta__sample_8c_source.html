41 /*
42  * Dequeue packets from DPDK queues
43  *
44  * @ingress_port [in]: port id for dequeue packets
45  */
46 static void process_packets(int ingress_port)
47 {
48  struct rte_mbuf *packets[PACKET_BURST];
49  int queue_index = 0;
50  int nb_packets;
51  int i;
52 
53  nb_packets = rte_eth_rx_burst(ingress_port, queue_index, packets, PACKET_BURST);
54 
55  /* Print received packets meta data */
56  if (nb_packets)
57  DOCA_LOG_INFO("Port %d: Received %d packets", ingress_port, nb_packets);
58  for (i = 0; i < nb_packets; i++) {
59  if (rte_flow_dynf_metadata_avail())
60  DOCA_LOG_INFO("Packet %d meta data %d", i, *RTE_FLOW_DYNF_METADATA(packets[i]));
61  }
62 }

64 /*
65  * Create DOCA Flow pipe with 5 tuple match, changeable set meta action, and forward RSS
66  *
67  * @port [in]: port of the pipe
68  * @pipe [out]: created pipe pointer
69  * @return: DOCA_SUCCESS on success and DOCA_ERROR otherwise
70  */
71 static doca_error_t create_rss_meta_pipe(struct doca_flow_port *port, struct doca_flow_pipe **pipe)
72 {
73  struct doca_flow_match match;
74  struct doca_flow_actions actions, *actions_arr[NB_ACTIONS_ARR];
75  struct doca_flow_fwd fwd;
76  struct doca_flow_fwd fwd_miss;
77  struct doca_flow_pipe_cfg *pipe_cfg;
78  uint16_t rss_queues[1];
79  doca_error_t result;
80 
81  memset(&match, 0, sizeof(match));
82  memset(&actions, 0, sizeof(actions));
83  memset(&fwd, 0, sizeof(fwd));
84  memset(&fwd_miss, 0, sizeof(fwd_miss));
85 
86  /* set mask value */
87  actions.meta.pkt_meta = UINT32_MAX;
88  actions_arr[0] = &actions;
89 
90  /* 5 tuple match */
91  match.parser_meta.outer_l4_type = DOCA_FLOW_L4_META_TCP;
92  match.parser_meta.outer_l3_type = DOCA_FLOW_L3_META_IPV4;
93  match.outer.l4_type_ext = DOCA_FLOW_L4_TYPE_EXT_TCP;
94  match.outer.l3_type = DOCA_FLOW_L3_TYPE_IP4;
95  match.outer.ip4.src_ip = 0xffffffff;
96  match.outer.ip4.dst_ip = 0xffffffff;
97  match.outer.tcp.l4_port.src_port = 0xffff;
98  match.outer.tcp.l4_port.dst_port = 0xffff;
99 
100  result = doca_flow_pipe_cfg_create(&pipe_cfg, port);
101  if (result != DOCA_SUCCESS) {
102  DOCA_LOG_ERR("Failed to create doca_flow_pipe_cfg: %s", doca_error_get_descr(result));
103  return result;
104  }
105 
106  result = set_flow_pipe_cfg(pipe_cfg, "RSS_META_PIPE", DOCA_FLOW_PIPE_BASIC, true);
107  if (result != DOCA_SUCCESS) {
108  DOCA_LOG_ERR("Failed to set doca_flow_pipe_cfg: %s", doca_error_get_descr(result));
109  goto destroy_pipe_cfg;
110  }
111  result = doca_flow_pipe_cfg_set_match(pipe_cfg, &match, NULL);
112  if (result != DOCA_SUCCESS) {
113  DOCA_LOG_ERR("Failed to set doca_flow_pipe_cfg match: %s", doca_error_get_descr(result));
114  goto destroy_pipe_cfg;
115  }
116  result = doca_flow_pipe_cfg_set_actions(pipe_cfg, actions_arr, NULL, NULL, NB_ACTIONS_ARR);
117  if (result != DOCA_SUCCESS) {
118  DOCA_LOG_ERR("Failed to set doca_flow_pipe_cfg actions: %s", doca_error_get_descr(result));
119  goto destroy_pipe_cfg;
120  }
121 
122  /* RSS queue - send matched traffic to queue 0 */
123  rss_queues[0] = 0;
124  fwd.type = DOCA_FLOW_FWD_RSS;
125  fwd.rss_type = DOCA_FLOW_RESOURCE_TYPE_NON_SHARED;
126  fwd.rss.queues_array = rss_queues;
127  fwd.rss.inner_flags = DOCA_FLOW_RSS_IPV4 | DOCA_FLOW_RSS_TCP;
128  fwd.rss.nr_queues = 1;
129 
130  fwd_miss.type = DOCA_FLOW_FWD_DROP;
131 
132  result = doca_flow_pipe_create(pipe_cfg, &fwd, &fwd_miss, pipe);
133 destroy_pipe_cfg:
134  doca_flow_pipe_cfg_destroy(pipe_cfg);
135  return result;
136 }

138 /*
139  * Add DOCA Flow pipe entry with example 5 tuple to match and set meta data value
140  *
141  * @pipe [in]: pipe of the entry
142  * @status [in]: user context for adding entry
143  * @return: DOCA_SUCCESS on success and DOCA_ERROR otherwise
144  */
145 static doca_error_t add_rss_meta_pipe_entry(struct doca_flow_pipe *pipe, struct entries_status *status)
146 {
147  struct doca_flow_match match;
148  struct doca_flow_actions actions;
149  struct doca_flow_pipe_entry *entry;
150  doca_error_t result;
151 
152  /* example 5-tuple to drop */
153  doca_be32_t dst_ip_addr = BE_IPV4_ADDR(8, 8, 8, 8);
154  doca_be32_t src_ip_addr = BE_IPV4_ADDR(1, 2, 3, 4);
155  doca_be16_t dst_port = rte_cpu_to_be_16(80);
156  doca_be16_t src_port = rte_cpu_to_be_16(1234);
157 
158  memset(&match, 0, sizeof(match));
159  memset(&actions, 0, sizeof(actions));
160 
161  match.outer.ip4.dst_ip = dst_ip_addr;
162  match.outer.ip4.src_ip = src_ip_addr;
163  match.outer.tcp.l4_port.dst_port = dst_port;
164  match.outer.tcp.l4_port.src_port = src_port;
165 
166  /* set meta value */
167  actions.meta.pkt_meta = DOCA_HTOBE32(10);
168  actions.action_idx = 0;
169 
170  result = doca_flow_pipe_add_entry(0, pipe, &match, &actions, NULL, NULL, 0, status, &entry);
171  if (result != DOCA_SUCCESS)
172  return result;
173 
174  return DOCA_SUCCESS;
175 }

177 /*
178  * Run flow_rss_meta sample
179  *
180  * @nb_queues [in]: number of queues the sample will use
181  * @return: DOCA_SUCCESS on success and DOCA_ERROR otherwise
182  */
183 doca_error_t flow_rss_meta(int nb_queues)
184 {
185  const int nb_ports = 2;
186  struct flow_resources resource = {0};
187  uint32_t nr_shared_resources[SHARED_RESOURCE_NUM_VALUES] = {0};
188  struct doca_flow_port *ports[nb_ports];
189  struct doca_dev *dev_arr[nb_ports];
190  uint32_t actions_mem_size[nb_ports];
191  struct doca_flow_pipe *pipe;
192  struct entries_status status;
193  int num_of_entries = 1;
194  doca_error_t result;
195  int port_id;
196 
197  result = init_doca_flow(nb_queues, "vnf,hws", &resource, nr_shared_resources);
198  if (result != DOCA_SUCCESS) {
199  DOCA_LOG_ERR("Failed to init DOCA Flow: %s", doca_error_get_descr(result));
200  return -1;
201  }
202 
203  memset(dev_arr, 0, sizeof(struct doca_dev *) * nb_ports);
204  ARRAY_INIT(actions_mem_size, ACTIONS_MEM_SIZE(nb_queues, num_of_entries));
205  result = init_doca_flow_ports(nb_ports, ports, true, dev_arr, actions_mem_size);
206  if (result != DOCA_SUCCESS) {
207  DOCA_LOG_ERR("Failed to init DOCA ports: %s", doca_error_get_descr(result));
208  doca_flow_destroy();
209  return result;
210  }
211 
212  for (port_id = 0; port_id < nb_ports; port_id++) {
213  memset(&status, 0, sizeof(status));
214 
215  result = create_rss_meta_pipe(ports[port_id], &pipe);
216  if (result != DOCA_SUCCESS) {
217  DOCA_LOG_ERR("Failed to create pipe: %s", doca_error_get_descr(result));
218  stop_doca_flow_ports(nb_ports, ports);
219  doca_flow_destroy();
220  return result;
221  }
222 
223  result = add_rss_meta_pipe_entry(pipe, &status);
224  if (result != DOCA_SUCCESS) {
225  DOCA_LOG_ERR("Failed to add entry: %s", doca_error_get_descr(result));
226  stop_doca_flow_ports(nb_ports, ports);
227  doca_flow_destroy();
228  return result;
229  }
230 
231  result = doca_flow_entries_process(ports[port_id], 0, DEFAULT_TIMEOUT_US, num_of_entries);
232  if (result != DOCA_SUCCESS) {
233  DOCA_LOG_ERR("Failed to process entries: %s", doca_error_get_descr(result));
234  stop_doca_flow_ports(nb_ports, ports);
235  doca_flow_destroy();
236  return result;
237  }
238 
239  if (status.nb_processed != num_of_entries || status.failure) {
240  DOCA_LOG_ERR("Failed to process entries");
241  stop_doca_flow_ports(nb_ports, ports);
242  doca_flow_destroy();
243  return DOCA_ERROR_BAD_STATE;
244  }
245  }
246 
247  DOCA_LOG_INFO("Wait few seconds for packets to arrive");
248  sleep(5);
249  for (port_id = 0; port_id < nb_ports; port_id++)
250  process_packets(port_id);
251 
252  result = stop_doca_flow_ports(nb_ports, ports);
253  doca_flow_destroy();
254  return result;
255 }
